# Generate above lists

Generate bullet list of readme files:

    find . -type f -iname 'readme*' | while read x; do
      echo "- [${x#./}](${x#./})"
    done

Generate bullet list of cheatsheets:

    find ./cheatsheets -type f -iname "*.md" | while read x; do
      echo "- [${x#./}](${x#./})"
    done
